# --file ../tests/circleofpts.txt --gverbose --use_atomic_conditions

INPUTS {
	px0 fl64 : (1.1, 5);
	py0 fl64 : (1.1, 6);
	cx0 fl64 : (2.0, 2.0);
	cy0 fl64 : (3.0, 3.0);
	radius fl64 : (10.0, 10.0);
}

OUTPUTS {
	final_radius ;
}


EXPRS {
	
	final_radius = radius ;

	dx0 = px0 - cx0;
	dy0 = py0 - cy0;

	dist2 = (dx0*dx0) + (dy0*dy0);

	if (dist2 > (final_radius*final_radius)) then
		dist = sqrt(dist2);
		newRadius = (final_radius + dist) * 0.5 ;
		final_radius = newRadius ;
	endif
}
